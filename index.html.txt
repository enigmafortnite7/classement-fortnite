  <script>
    async function chargerDonnees() {
      const url = "https://docs.google.com/spreadsheets/d/1iud_FJPMdjSR6zCYIlSuRf7GSdeYnPkuT2SPXu2KRhQ/gviz/tq?tqx=out:json";

      const response = await fetch(url);
      const text = await response.text();

      // Google renvoie un JSON encapsulé → nettoyage
      const json = JSON.parse(text.substr(47).slice(0, -2));
      const rows = json.table.rows;

      let equipes = [];

      // ⚡ On commence à la ligne 6 (ignorer les 5 premières)
      for (let i = 5; i < rows.length; i += 2) {
        const row1 = rows[i]?.c;
        const row2 = rows[i+1]?.c;

        if (row1 && row2) {
          equipes.push({
            rang: row1[15]?.v || "",
            joueur1: row1[3]?.v || "",
            joueur2: row2[3]?.v || "",
            score: row1[14]?.v || ""
          });
        }
      }

      equipes.sort((a, b) => parseInt(a.rang) - parseInt(b.rang));

      const container = document.getElementById("resultats");
      container.innerHTML = "";

      equipes.forEach(row => {
        let rankText;
        switch (parseInt(row.rang)) {
          case 1: rankText = "1st"; break;
          case 2: rankText = "2nd"; break;
          case 3: rankText = "3rd"; break;
          default: rankText = row.rang + "th"; break;
        }

        let specialClass = "";
        if (parseInt(row.rang) === 1) specialClass = "first";
        if (parseInt(row.rang) === 2) specialClass = "second";
        if (parseInt(row.rang) === 3) specialClass = "third";

        container.innerHTML += `
          <div class="result-row ${specialClass}">
            <div class="rank">${rankText}</div>
            <div class="player">${row.joueur1}</div>
            <div class="player">${row.joueur2}</div>
            <div class="score">${row.score}</div>
          </div>
        `;
      });
    }

    chargerDonnees();
    setInterval(chargerDonnees, 10000); // refresh auto toutes les 10 sec
  </script>
